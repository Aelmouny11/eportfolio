```python
def swap_tuple(Tuple1,Tuple2):
    list1=list(Tuple1)
    list2=list(Tuple2)
    list1,list2=list2,list1
    return tuple(list2),tuple(list1)
swap_tuple(('hgfgf','gfdfdf'),('2','33'))
```

```python
def find_replace(Tuple,target,new_value):
    for item in Tuple:

                return 0;
                
def modifier(Tuple,target,new_value):
```
